# SQL Queries for Testing

SELECT * FROM CUSTOMER;

SELECT * FROM LOAN_DETAILS;

SELECT customer_name FROM CUSTOMER;

SELECT loan_amount FROM LOAN_DETAILS;

SELECT * FROM LOAN_DETAILS WHERE loan_status='Approved';

SELECT * FROM LOAN_DETAILS WHERE loan_status='Rejected';

SELECT COUNT(*) FROM CUSTOMER;

SELECT COUNT(*) FROM LOAN_DETAILS;

SELECT customer_name, loan_amount FROM CUSTOMER
JOIN LOAN_DETAILS
ON CUSTOMER.customer_id = LOAN_DETAILS.customer_id;

SELECT * FROM LOAN_DETAILS ORDER BY loan_amount DESC;

SELECT * FROM LOAN_DETAILS WHERE loan_amount > 50000;

SELECT * FROM LOAN_DETAILS WHERE loan_amount < 50000;

SELECT DISTINCT loan_status FROM LOAN_DETAILS;

SELECT MAX(loan_amount) FROM LOAN_DETAILS;

SELECT MIN(loan_amount) FROM LOAN_DETAILS;
